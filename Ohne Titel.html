                return;

            };

            paginationContainer.classList.remove('hidden');


            for (let i = 1; i <= totalPages; i++) {

                const button = document.createElement('button');

                button.textContent = i;

                button.className = `pagination-btn px-4 py-2 border rounded-lg transition hover:bg-gray-100`;

                if (i === currentPage) {

                    button.classList.add('active');

                }

                button.addEventListener('click', () => {

                    currentPage = i;

                    renderResults();

                    renderPagination(); 

                });

                paginationContainer.appendChild(button);

            }

        }


        function renderResults() {

            resultsContainer.innerHTML = ''; 


            const sortedModels = [...currentFilteredModels].sort((a, b) => (b.featured ? 1 : 0) - (a.featured ? 1 : 0));

            

            const startIndex = (currentPage - 1) * itemsPerPage;

            const endIndex = startIndex + itemsPerPage;

            const paginatedModels = sortedModels.slice(startIndex, endIndex);


            if (paginatedModels.length === 0 && currentPage === 1) {

                noResultsMessage.classList.remove('hidden');

            } else {

                noResultsMessage.classList.add('hidden');

            }


            paginatedModels.forEach(model => {

                const card = document.createElement('div');

                const modelLang = model[currentLang];

                card.className = `card bg-white p-6 rounded-2xl shadow-md border flex flex-col relative overflow-hidden ${model.featured ? 'featured-card border-2' : 'border-gray-200'}`;

                card.dataset.modelName = model.name;

                

                const beginnerBadgeHTML = model.beginnerFriendly ? `<span class="bg-green-100 text-green-800 text-xs font-semibold px-2.5 py-1 rounded-full flex items-center gap-1.5"><i data-lucide="star" class="w-3 h-3"></i>${translations[currentLang].beginnerFriendly}</span>` : '';


                card.innerHTML = `

                    <div class="flex-grow">

                         <div class="flex justify-between items-start mb-3">

                            <h3 class="text-xl font-bold text-gray-900 pr-4">${model.name}</h3>

                            ${beginnerBadgeHTML}

                        </div>

                        <div class="flex items-center gap-4 text-sm text-gray-500 mb-4">

                            <span class="flex items-center gap-1.5 bg-blue-100 text-blue-800 p-1.5 rounded-md text-xs"><i data-lucide="tag" class="w-3 h-3"></i>${modelLang.category}</span>

                            <span class="flex items-center gap-1.5 bg-yellow-100 text-yellow-800 p-1.5 rounded-md text-xs"><i data-lucide="dollar-sign" class="w-3 h-3"></i>${translations[currentLang].pricing[model.pricing]}</span>

                        </div>

                        <p class="text-gray-600">${modelLang.description}</p>

                    </div>

                `;

                

                card.addEventListener('click', () => {

                    const selectedModel = allAiModels.find(m => m.name === model.name);

                    if (selectedModel) {

                        showDetailView(selectedModel);

                    }

                });


                resultsContainer.appendChild(card);

            });

            lucide.createIcons();

        }


        function filterAndRender() {

            const selectedCategory = categorySelect.value;

            const searchTerm = searchInput.value.toLowerCase().trim();


            currentFilteredModels = allAiModels.filter(model => {

                const modelLang = model[currentLang];

                const categoryMatch = selectedCategory === 'all' || model.de.category === selectedCategory;

                

                if (!searchTerm) return categoryMatch;


                const searchableText = [

                    model.name.toLowerCase(),

                    modelLang.description.toLowerCase(),

                    ...modelLang.useCases.map(uc => uc.toLowerCase())

                ].join(' ');

                const searchMatch = searchableText.includes(searchTerm);


                return categoryMatch && searchMatch;

            });

            

            currentPage = 1; 

            renderResults();

            renderPagination();

        }


        // --- EVENT LISTENERS ---

        categorySelect.addEventListener('change', filterAndRender);

        searchInput.addEventListener('input', filterAndRender);

        backToListBtn.addEventListener('click', showMainView);

        langDeBtn.addEventListener('click', () => updateContent('de'));

        langEnBtn.addEventListener('click', () => updateContent('en'));


        // --- INITIAL RENDER ---

        document.getElementById('copyright-year').textContent = new Date().getFullYear();

        lucide.createIcons();

        updateContent(currentLang); // This will also call filterAndRender
